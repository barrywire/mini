# Identifiers
<identifier> ::= <char> { <char>* | '_' | <char> }
<char> ::= <letter_uppercase> | <letter_lowercase> | <letter_titlecase> | <letter_camelcase>
<letter_uppercase> ::= "A" ... "Z"
<letter_lowercase> ::= "a" ... "z"
